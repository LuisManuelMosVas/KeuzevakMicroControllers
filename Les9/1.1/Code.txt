int inByte = 0; 
const int ledPin = 10;
const int time_base = 100; 
const int dash = 3*time_base; 

int audio8 =8;
int note = 1200; 
int dotLen = 100; 
int dashLen = dotLen * 3;    
int Paus = dotLen;  
int Spaces = dotLen * 3;     
int wordPause = dotLen *7;
void MorseDot()
{
	
	tone(audio8, note, dotLen);	
	delay(dotLen);             	
}

void MorseDash()
{
	
	tone(audio8, note, dashLen);	
	delay(dashLen);               
}
void LightsOff(int delayTime)
{
	
	noTone(audio8);	       	   
	delay(delayTime);           
}
void setup() {
	
	
	Serial.begin(9600);
	pinMode(ledPin,OUTPUT);} 

void loop() {
	
	
	
	if (Serial.available() > 0){ 
		inByte = Serial.read();}
	
	
	if (inByte == 'a'){
		MorseDot();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		delay(3*time_base);} 
	
	// Morse code for b (dash dot dot dot)
	if (inByte == 'b'){
		MorseDash();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for c (dash dot dash dot)
	if (inByte == 'c'){
		MorseDash();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for d (dash dot dot)
	if (inByte == 'd'){
		MorseDash();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		delay(3*time_base);}
	
	//Morse code for e (dot)
	if (inByte == 'e') {
		MorseDot();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for f (dot dot dash dot)
	if (inByte == 'f'){
		MorseDot();
		digitalWrite(ledPin,HIGH);
		delay(time_base);
		digitalWrite(ledPin,LOW);
		delay(time_base);
		digitalWrite(ledPin,HIGH);
		delay(time_base);
		MorseDash();
		digitalWrite(ledPin,LOW);
		delay(time_base);
		digitalWrite(ledPin,HIGH);
		delay(dash);
		digitalWrite(ledPin,LOW);
		delay(time_base);
		MorseDot();
		digitalWrite(ledPin,HIGH);
		delay(time_base);
		digitalWrite(ledPin,LOW);
		delay(3*time_base);}
	
	// Morse code for g (dash dash dot)
	if (inByte == 'g'){
		MorseDash();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for h (dot dot dot dot)
	if (inByte == 'h'){
		MorseDot();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for i (dot dot)
	if (inByte == 'i'){
		MorseDot();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for j (dot dash dash dash)
	if (inByte == 'j'){
		MorseDot();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for k (dash dot dash)
	if (inByte == 'k'){
		MorseDash();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for l (dot dash dot dot)
	if (inByte == 'l'){
		MorseDot();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for m (dash dash)
	if (inByte == 'm'){
		MorseDash();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for n (dash dot)
	if (inByte == 'n'){
		MorseDash();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for o (dash dash dash)
	if (inByte == 'o'){
		MorseDash();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for p (dot dash dash dot)
	if (inByte == 'p'){
		MorseDot();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for q (dash dash dot dash)
	if (inByte == 'q'){
		MorseDash();
		digitalWrite(ledPin,HIGH);
		delay(dash);
		digitalWrite(ledPin,LOW);
		delay(time_base);
		MorseDash();
		digitalWrite(ledPin,HIGH);
		delay(dash);
		
		digitalWrite(ledPin,LOW);
		delay(time_base);
		MorseDot();
		digitalWrite(ledPin,HIGH);
		delay(time_base);
		digitalWrite(ledPin,LOW);
		delay(time_base);
		digitalWrite(ledPin,HIGH);
		delay(dash);
		MorseDash();
		digitalWrite(ledPin,LOW);
		delay(3*time_base);}
	
	// Morse code for r (dot dash dot)
	if (inByte == 'r'){
		MorseDot();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for s (dot dot dot)
	if (inByte == 's'){
		MorseDot();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for t (dash)
	if (inByte =='t'){
		MorseDash();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for u (dot dot dash)
	if (inByte =='u'){
		
		MorseDot();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for v (dot dot dot dash)
	if (inByte == 'v'){
		MorseDot();
		digitalWrite(ledPin,HIGH);
		delay(time_base);
		digitalWrite(ledPin,LOW);
		MorseDot();
		delay(time_base);
		digitalWrite(ledPin,HIGH);
		delay(time_base);
		MorseDot();
		digitalWrite(ledPin,LOW);
		delay(time_base);
		digitalWrite(ledPin,HIGH);
		delay(time_base);
		digitalWrite(ledPin,LOW);
		delay(time_base);
		MorseDash();
		digitalWrite(ledPin,HIGH);
		delay(dash);
		digitalWrite(ledPin,LOW);
		delay(3*time_base);}
	
	// Morse code for w (dot dash dash)
	if (inByte == 'w'){
		MorseDot();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for x (dash dot dot dash)
	if (inByte == 'x'){
		MorseDash();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for y (dash dot dash dash)
	if (inByte == 'y'){
		MorseDash();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		delay(dash);
		MorseDash();
		digitalWrite(ledPin,LOW);
		delay(3*time_base);}
	
	
	// Morse code for z (dash dash dot dot)
	if (inByte == 'z'){
		MorseDash();
		LightsOff(Paus);
		MorseDash();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		MorseDot();
		LightsOff(Paus);
		delay(3*time_base);}
	
	// Morse code for un space (7 dots)
	if (inByte == ' '){
		digitalWrite(ledPin,LOW);
		delay(7*time_base);}
	
	
	
}
